## 执行脚本

```bash
for SEED in 1 2 3; do
    MODEL="checkpoints/orch_grpo_7b_seed${SEED}/final"

    # ── Cheap worker pool (全部 4 tracks) ──
    for TRACK in qa code gpqa livecode; do
        python eval/eval_orchestrator.py \
            --model_path $MODEL \
            --data_path data/test_${TRACK}.jsonl \
            --worker_pool cheap \
            --api_base $API_BASE --api_key $API_KEY \
            --output eval/results/orch_grpo_7b_seed${SEED}_cheap_${TRACK}.json
    done

    # ── Matched worker pool (仅 GPQA + LiveCodeBench) ──
    for TRACK in gpqa livecode; do
        python eval/eval_orchestrator.py \
            --model_path $MODEL \
            --data_path data/test_${TRACK}.jsonl \
            --worker_pool matched \
            --api_base $API_BASE --api_key $API_KEY \
            --output eval/results/orch_grpo_7b_seed${SEED}_matched_${TRACK}.json
    done
done
```
