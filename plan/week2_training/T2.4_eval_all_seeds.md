## 执行脚本

```bash
for SEED in 1 2 3; do
    MODEL="checkpoints/orch_grpo_7b_seed${SEED}/final"

    # QA Track (NQ + TriviaQA + PopQA + HotpotQA + 2Wiki + MuSiQue)
    python eval/eval_orchestrator.py \
        --model_path $MODEL \
        --data_path data/test_qa.jsonl \
        --api_base $API_BASE --api_key $API_KEY \
        --output eval/results/orch_grpo_7b_seed${SEED}_qa.json

    # Code Track
    python eval/eval_orchestrator.py \
        --model_path $MODEL \
        --data_path data/test_code.jsonl \
        --output eval/results/orch_grpo_7b_seed${SEED}_code.json

    # GPQA Track (如果评估脚本支持)
    python eval/eval_orchestrator.py \
        --model_path $MODEL \
        --data_path data/test_gpqa.jsonl \
        --output eval/results/orch_grpo_7b_seed${SEED}_gpqa.json
done
```
