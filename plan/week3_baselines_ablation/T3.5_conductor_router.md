    --model_path path/to/router_r1_checkpoint \
    --question_file ../OrchestratorR1/data/test_qa.jsonl \
    --api_base $API_BASE --api_key $API_KEY \
    --output ../OrchestratorR1/eval/results/router_r1_reproduced.json
